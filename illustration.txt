#1.MySQL offset表创建语句
create table kafka_offset(
    topic varchar(200),
    partition int,
    offset int
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

#2.MySQL error_log表创建语句
create table error_log(
    log_time varchar(200) primary key,
    log_count int
)ENGINE=InnoDB DEFAULT CHARSET=utf8;
